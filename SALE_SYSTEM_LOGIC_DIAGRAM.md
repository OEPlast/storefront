### Random Assignment

```typescript
function getBoughtCountPercentage(): number {
  const rand = Math.random();
  if (rand < 0.1) return 0.9;  // 10% chance
  if (rand < 0.5) return 0.5;  // 40% chance (0.1 to 0.5)
  return 0.1;                  // 50% chance (0.5 to 1.0)
}
```
